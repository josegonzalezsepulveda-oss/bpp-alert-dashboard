Queries BigQuery — Panel Alertas BPP

Ejecutar cada query en BigQuery, descargar como JSON y reemplazar la sección DATA en el dashboard.

SELECT COUNT(DISTINCT bpp.SHP_SHIPMENT_ID) AS total_alertas, COUNT(DISTINCT inv.TRS_CASE_ID) AS alertas_con_caso, COUNT(DISTINCT bpp.SHP_SHIPMENT_ID) - COUNT(DISTINCT inv.TRS_CASE_ID) AS alertas_sin_caso, ROUND(COUNT(DISTINCT inv.TRS_CASE_ID) * 100.0 / NULLIF(COUNT(DISTINCT bpp.SHP_SHIPMENT_ID), 0), 1) AS tasa_conversion_pct, ROUND(SUM(bpp.BPP_CASHOUT_USD), 2) AS monto_total_riesgo_usd, ROUND(SUM(CASE WHEN inv.TRS_CASE_ID IS NULL THEN bpp.BPP_CASHOUT_USD END), 2) AS monto_sin_caso_usd, COUNTIF(bpp.REPEAT_BEHAVIOR_FLAG = TRUE) AS alertas_reincidentes, ROUND(AVG(CASE WHEN inv.TRS_CASE_ID IS NOT NULL THEN DATE_DIFF(inv.TRS_CASE_CREATED_DATE, DATE(bpp.AUD_INS_DTTM), DAY) END), 1) AS tat_promedio_dias FROM `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS_SHIPMENTS_BPP_BEHAVIOR` bpp LEFT JOIN `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS` inv ON bpp.SHP_SHIPMENT_ID = inv.TRS_CASE_BPP_SHIPMENT_ID;
SELECT bpp.SIT_SITE_ID AS site, COUNT(DISTINCT bpp.SHP_SHIPMENT_ID) AS total_alertas, COUNT(DISTINCT inv.TRS_CASE_ID) AS con_caso, COUNT(DISTINCT bpp.SHP_SHIPMENT_ID) - COUNT(DISTINCT inv.TRS_CASE_ID) AS sin_caso, ROUND(COUNT(DISTINCT inv.TRS_CASE_ID) * 100.0 / NULLIF(COUNT(DISTINCT bpp.SHP_SHIPMENT_ID), 0), 1) AS tasa_conversion_pct, ROUND(SUM(bpp.BPP_CASHOUT_USD), 2) AS monto_riesgo_usd, ROUND(SUM(CASE WHEN inv.TRS_CASE_ID IS NOT NULL THEN inv.TRS_ANALYZED_AMOUNT END), 2) AS monto_analizado_usd FROM `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS_SHIPMENTS_BPP_BEHAVIOR` bpp LEFT JOIN `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS` inv ON bpp.SHP_SHIPMENT_ID = inv.TRS_CASE_BPP_SHIPMENT_ID GROUP BY 1 ORDER BY total_alertas DESC;
SELECT COALESCE(bpp.CAUSA_BPP, 'Sin Causa') AS causa_bpp, COUNT(DISTINCT bpp.SHP_SHIPMENT_ID) AS total_alertas, COUNT(DISTINCT inv.TRS_CASE_ID) AS con_caso, COUNT(DISTINCT bpp.SHP_SHIPMENT_ID) - COUNT(DISTINCT inv.TRS_CASE_ID) AS sin_caso, ROUND(COUNT(DISTINCT inv.TRS_CASE_ID) * 100.0 / NULLIF(COUNT(DISTINCT bpp.SHP_SHIPMENT_ID), 0), 1) AS tasa_conversion_pct, ROUND(SUM(bpp.BPP_CASHOUT_USD), 2) AS monto_riesgo_usd FROM `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS_SHIPMENTS_BPP_BEHAVIOR` bpp LEFT JOIN `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS` inv ON bpp.SHP_SHIPMENT_ID = inv.TRS_CASE_BPP_SHIPMENT_ID GROUP BY 1 ORDER BY total_alertas DESC;
SELECT COALESCE(inv.TRS_ANALYSIS_TYPE, 'Sin Tipo') AS tipo_analisis, COALESCE(inv.TRS_CASE_STATUS, 'Sin Estado') AS estado_caso, COUNT(DISTINCT inv.TRS_CASE_ID) AS total_casos, ROUND(SUM(inv.TRS_ANALYZED_AMOUNT), 2) AS monto_analizado_usd FROM `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS` inv WHERE inv.TRS_CASE_BPP_SHIPMENT_ID IS NOT NULL AND (inv.TRS_CASE_DELETE_FLAG IS FALSE OR inv.TRS_CASE_DELETE_FLAG IS NULL) GROUP BY 1, 2 ORDER BY total_casos DESC;
SELECT FORMAT_DATE('%Y-%m', bpp.DATEPARAMETER) AS mes, FORMAT_DATE('%b %Y', bpp.DATEPARAMETER) AS mes_label, bpp.SIT_SITE_ID AS site, COUNT(DISTINCT bpp.SHP_SHIPMENT_ID) AS total_alertas, COUNT(DISTINCT inv.TRS_CASE_ID) AS con_caso, ROUND(COUNT(DISTINCT inv.TRS_CASE_ID) * 100.0 / NULLIF(COUNT(DISTINCT bpp.SHP_SHIPMENT_ID), 0), 1) AS tasa_conversion_pct, ROUND(SUM(bpp.BPP_CASHOUT_USD), 2) AS monto_usd FROM `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS_SHIPMENTS_BPP_BEHAVIOR` bpp LEFT JOIN `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS` inv ON bpp.SHP_SHIPMENT_ID = inv.TRS_CASE_BPP_SHIPMENT_ID WHERE bpp.DATEPARAMETER >= DATE_TRUNC(DATE_SUB(CURRENT_DATE(), INTERVAL 6 MONTH), MONTH) GROUP BY 1, 2, 3 ORDER BY 1, 3;
SELECT bpp.SIT_SITE_ID AS site, COUNTIF(bpp.CASE_BPP_EXPIRATION_DAY <= 0) AS vencidas, COUNTIF(bpp.CASE_BPP_EXPIRATION_DAY BETWEEN 1 AND 3) AS vence_3_dias, COUNTIF(bpp.CASE_BPP_EXPIRATION_DAY BETWEEN 4 AND 7) AS vence_7_dias, COUNTIF(bpp.CASE_BPP_EXPIRATION_DAY > 7) AS vigentes, ROUND(SUM(CASE WHEN bpp.CASE_BPP_EXPIRATION_DAY <= 0 THEN bpp.BPP_CASHOUT_USD END), 2) AS monto_vencido_usd FROM `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS_SHIPMENTS_BPP_BEHAVIOR` bpp LEFT JOIN `meli-bi-data.WHOWNER.BT_LP_TRS_INVESTIGATIONS` inv ON bpp.SHP_SHIPMENT_ID = inv.TRS_CASE_BPP_SHIPMENT_ID WHERE inv.TRS_CASE_ID IS NULL GROUP BY 1 ORDER BY vencidas DESC;
